import { graphql } from "gatsby"

export const posts = graphql`
  query Posts {
    allMdx(filter: { frontmatter: { type: { eq: "post" } } }) {
      nodes {
        excerpt
        timeToRead
        wordCount {
          paragraphs
          sentences
          words
        }
        frontmatter {
          date
          title
          path
          type
        }
        tableOfContents
      }
    }
  }
`

<div>
  {props.data.allMdx.nodes.map(({ frontmatter }) => (
    <div key={frontmatter.title}>
      <a href={frontmatter.path}>{frontmatter.path}</a>
      <span>{frontmatter.title}</span>
      <span>{frontmatter.date}</span>
    </div>
  ))}
</div>
